489     /**
490      * The and operator takes one or more sub-expressions each with a base-type
491      * boolean and single cardinality. The result is a single boolean which is
492      * if all sub-expressions are true and false if any of them are false.
493      *
494      * @access public
495      * @author Cedric Alfonsi, <cedric.alfonsi@tudor.lu>
496      * @return boolean
497      */
498     , and : function(options){
499         var returnValue = null;
500         options = this.checkOptions(options);
501         
502         var paramCount = this.and.arguments.length;
503 
504         for (var i = 1; i < paramCount; i++) {
505             var subExp = this.and.arguments[i];
506             var matchingSubExp = TAO_MATCHING.VariableFactory.toBooleanBaseType (subExp);
507             
508             if (matchingSubExp == null) {
509                 throw new Error ("TtaoQTI_models_classes_Matching_Matching::and an error occured : The ["+i+"]expression passed ["+matchingSubExp+"] to the operator has to be a valid boolean expression with single cardinality");
510             } else {
511                 if (matchingSubExp.isNull()){
512                     returnValue = null;
513                     break;
514                 }else{
515                     if (returnValue === null){
516                         returnValue = matchingSubExp.getValue();
517                     } else {
518                         returnValue = returnValue && matchingSubExp.getValue();
519                     }
520                 }
521             }
522         }
523         
524         return returnValue;
525     }

527     /**
528      * Create a variable functions of the arguments.
529      * Create either scalar or container from the value :
530      * createVariable (null, 3.1415);
531      * createVariable (null, Array ("TAO", "Test Assist par Ordinateur"))
532      * Create container following the options.type and the arguments of the
533      * createVariable (Array("type"=>"list"), "TAO", "Test Assist par
534      *
535      * @access public
536      * @author Cedric Alfonsi, <cedric.alfonsi@tudor.lu>
537      * @param  array options
538      * @return taoQTI_models_classes_Matching_Variable
539      */
540     , createVariable : function (options, type) {
541         var returnValue = null;
542         options = this.checkOptions(options);
543                 
544         // Type undefined, we are in the case of baseTypeVariable creation (cardinality single)
545         if (typeof(options.type) == 'undefined' ) {
546             returnValue = TAO_MATCHING.VariableFactory.create (this.createVariable.arguments[1]);
547         }
548         else
549         {
550             switch (options.type){
551                 case 'integer':
552                 case 'float':
553                 case 'string':
554                 case 'boolean':
555                     // In all the base type cases create a variable with the first found argument
556                     returnValue = TAO_MATCHING.VariableFactory.create (this.createVariable.arguments[1]);
557                     break;
558                     
559                 case 'tuple':
560                     var values = {};
561                     var a = 0;
562                     for (var i = 1; i < this.createVariable.arguments.length; ++i, ++a) {
563                         values[a] = this.createVariable.arguments[i];
564                     }
565                     //var tuple = new TAO_MATCHING.Tuple (values);
566                     //console.log (tuple);
567                     
568                     returnValue = TAO_MATCHING.VariableFactory.create (values);
569                     break;
570                     
571                 case 'list':
572                     var values = [];
573                     var a = 0;
574                     for (var i = 1; i < this.createVariable.arguments.length; ++i, ++a) {
575                         values.push (this.createVariable.arguments[i]);
576                     }
577                     returnValue = TAO_MATCHING.VariableFactory.create (values);
578                     break;
579                     
580                 case 'default':
581                     throw new Error ('TAO_MATCHING.createVariable : type unknown ['+options.type+']');
582             }
583         }
584         
585         //console.log(returnValue.toJSon());
586         
587         return returnValue;
588     }

591     /**
592      * The contains function takes two sub-expressions. The first one has a
593      * - either list or tuple. The second one could have any base type and could
594      * the same cardinality than the first expression or it could have a single
595      * The result is a single boolean with a value of true if the container
596      * by the first sub-expression contains the value given by the second
597      * and false if it doesn't. Note that the contains operator works
598      * depending on the cardinality of the two sub-expressions. For unordered
599      * the values are compared without regard for ordering, for example, [A,B,C]
600      * [C,A]. Note that [A,B,C] does not contain [B,B] but that [A,B,B,C] does.
601      * ordered containers the second sub-expression must be a strict
602      * within the first. In other words, [A,B,C] does not contain [C,A] but it
603      * contain [B,C].
604      *
605      * @access public
606      * @author Cedric Alfonsi, <cedric.alfonsi@tudor.lu>
607      * @param  options
608      * @param  expr1
609      * @param  expr2
610      * @return taoQTI_models_classes_Matching_bool
611      */
612     , contains : function (options, expr1, expr2){
613         var returnValue = null;
614         options = this.checkOptions (options);
615                 
616         if (!TAO_MATCHING.Variable.isCollection(expr1)){
617             throw new Error ("TtaoQTI_models_classes_Matching_Matching::contains \
618             an error occured : The operator contains as first argument an expression of type Collection");
619         }
620         
621         returnValue = expr1.contains(expr2, options);
622         
623         return returnValue;
624     }

626     /**
627      * The divide operator takes 2 sub-expressions which both 
628      * have single cardinality and numerical base-types. The
629      * result is a single float that corresponds to the first 
630      * expression divided by the second expression. If either
631      * of the sub-expressions is NULL then the operator 
632      * results in NULL.
633      *
634      * @access public
635      * @author Cedric Alfonsi, <cedric.alfonsi@tudor.lu>
636      * @param  options
637      * @param  expr1
638      * @param  expr2
639      */
640     , divide : function (options, expr1, expr2){
641         var returnValue = null;
642         
643         // IF the first expression is not a numerical base type
644         var matchingExpr1 = TAO_MATCHING.VariableFactory.toNumericBaseType (expr1);
645         if (matchingExpr1 == null) {
646             throw new Error ("TAO_MATCHING.Matching::substract an error occured : The first expression passed ["+expr1+"] to the operator has to be a valid numerical expression with single cardinality");
647         }
648 
649         // IF the first expression is not a numerical base type
650         var matchingExpr2 = TAO_MATCHING.VariableFactory.toNumericBaseType (expr2);
651         if (matchingExpr2 == null) {
652             throw new Error ("TAO_MATCHING.Matching::substract an error occured : The second expression passed ["+expr2+"] to the operator has to be a valid numerical expression with single cardinality");
653         }
654         
655         if (matchingExpr1.getValue() == null || matchingExpr2.getValue() == null || matchingExpr2.getValue() == 0){
656             returnValue = null;
657         } else {
658             returnValue = matchingExpr1.getValue() / matchingExpr2.getValue();
659         }
660         
661         return returnValue;
662     }

664     /**
665      * The equal operator takes two sub-expressions which must both have single
666      * and have a numerical base-type. The result is a single boolean with a
667      * of true if the two expressions are numerically equal and false if they
668      * not.
669      *
670      * @access public
671      * @author Cedric Alfonsi, <cedric.alfonsi@tudor.lu>
672      * @param  expr1
673      * @param  expr2
674      * @return boolean
675      */
676     , equal : function(options, expr1, expr2){
677         var result = null;
678         options = this.checkOptions(options);
679 
680         var value1 = null;
681         var value2 = null;
682         
683         // The first expression is a Matching BaseTypeVariable
684         if (expr1 instanceof TAO_MATCHING.BaseTypeVariable) {
685             value1 = expr1.getValue();
686         }
687         // The first expression is not a Matching BaseTypeVariable
688         else {
689             if (!TAO_MATCHING.Variable.isScalar(expr1)) {
690                 throw new Error('TAO_MATCHING.matching::equal an error occured : the first argument ['+expr1+'] must be a scalar');
691             }
692             else {
693                 value1 = expr1;
694             }
695         }
696         
697         // The second expression is a Matching BaseTypeVariable
698         if (expr2 instanceof TAO_MATCHING.BaseTypeVariable) {
699             value2 = expr2.getValue();
700         }
701         // The second expression is not a Matching BaseTypeVariable
702         else {
703             if (!TAO_MATCHING.Variable.isScalar(expr2)) {
704                 throw new Error('TAO_MATCHING.matching::equal an error occured : the second argument ['+expr2+'] must be a scalar');
705             }
706             else {
707                 value2 = expr2;

1350         var matchingExpr2 = TAO_MATCHING.VariableFactory.toNumericBaseType (expr2);
1351         if (matchingExpr2 == null) {
1352             throw new Error ("TAO_MATCHING.Matching::substract an error occured : The second expression passed ["+expr2+"] to the operator has to be a valid numerical expression with single cardinality");
1353         }
1354         
1355         if (matchingExpr1.getValue() == null || matchingExpr2.getValue() == null){
1356             returnValue = null;
1357         } else {
1358             returnValue = matchingExpr1.getValue() - matchingExpr2.getValue();
1359         }
1360         
1361         return returnValue;
1362     }

1365      * The sum operator takes 1 or more sub-expressions which all
1366      * have single cardinality and have numerical base-types. The
1367      * result is a single float or, if all sub-expressions are of integer
1368      * type, a single integer that corresponds to the sum of the
1369      * numerical values of the sub-expressions.

1371      * @access public

1373      * @param  options
1374      */
1375     , sum : function (options){
1376         var returnValue = 0;
1377         
1378         options = this.checkOptions(options);
1379         var paramCount = this.sum.arguments.length;
1380         
1381         for (var i = 1; i < paramCount; i++) {
1382             var subExp = this.sum.arguments[i];
1383             var matchingSubExp = TAO_MATCHING.VariableFactory.toNumericBaseType (subExp);
1384             if (matchingSubExp == null){

1386             } else if (matchingSubExp.isNull()){
1387                 returnValue = null;
1388                 break;
1389             } else {
1390                 returnValue += matchingSubExp.getValue();
1391             }
1392         }
1393
